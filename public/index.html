// Analyser les données pour détecter les fraudeurs
for (let i = 1; i < data.length; i++) {
  const ligne = data[i];
  
  // Extraire les données pertinentes
  const dateTime = ligne[dateTimeIndex];
  const action = ligne[actionIndex];
  const memberId = ligne[memberIdIndex];
  const remoteType = ligne[remoteTypeIndex];
  
  // Si l'ID du membre n'est pas initialisé dans notre dictionnaire
  if (!actionsMembres[memberId]) {
    actionsMembres[memberId] = {
      derniereAction: null,
      derniereDate: null,
      historique: [],
      consecutiveEntries: 0,
      nombreEntrees: 0
    };
  }
  
  // Mettre à jour le suivi selon l'action
  if (action === "GrantedInside") {
    // Incrémente le compteur d'entrées
    actionsMembres[memberId].nombreEntrees++;
    
    // Détection de comportements suspects
    if (actionsMembres[memberId].derniereAction === "GrantedInside") {
      actionsMembres[memberId].consecutiveEntries++;
      
      // Si deux GrantedInside consécutifs, c'est un fraudeur potentiel
      if (actionsMembres[memberId].consecutiveEntries >= 1) {
        fraudeursIds.add(memberId);
      }
    } else {
      actionsMembres[memberId].consecutiveEntries = 0;
    }
  }
}

Ce script Google Apps Script analyse les entrées dans un bâtiment pour détecter des comportements frauduleux. Il identifie les membres qui entrent plusieurs fois sans sortir, révélant une potentielle fraude par partage de badge. L'algorithme surveille les séquences d'entrées consécutives et le déséquilibre entre entrées/sorties pour générer automatiquement un rapport des fraudeurs potentiels.
